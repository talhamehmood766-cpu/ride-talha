    <script>
        // Theme Toggle
        const toggle = document.getElementById('theme-toggle');
        const icon = toggle.querySelector('i');
        const html = document.documentElement;

        function setTheme(isDark) {
            if (isDark) {
                html.classList.add('dark');
                icon.classList.replace('fa-moon', 'fa-sun');
            } else {
                html.classList.remove('dark');
                icon.classList.replace('fa-sun', 'fa-moon');
            }
            localStorage.setItem('theme', isDark ? 'dark' : 'light');
        }

        toggle.addEventListener('click', () => {
            setTheme(!html.classList.contains('dark'));
        });

        // Load saved theme
        if (localStorage.getItem('theme') === 'dark' || 
            (!localStorage.getItem('theme') && window.matchMedia('(prefers-color-scheme: dark)').matches)) {
            setTheme(true);
        }

        // Scroll Animations
        const fadeElements = document.querySelectorAll('.fade-in');
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('visible');
                }
            });
        }, { threshold: 0.1 });

        fadeElements.forEach(el => observer.observe(el));

        // Mobile Menu (if needed)
        document.getElementById('mobile-menu-btn')?.addEventListener('click', () => {
            // toggle mobile menu logic here if expanded
        });

        // Booking Form Submit
        document.getElementById('booking-form')?.addEventListener('submit', e => {
            e.preventDefault();
            const modal = document.getElementById('success-modal');
            modal.classList.remove('hidden');
            setTimeout(() => {
                modal.querySelector('div').classList.remove('scale-95', 'opacity-0');
                modal.querySelector('div').classList.add('scale-100', 'opacity-100');
            }, 100);
        });

        // Simple Testimonial Carousel (expandable)
        const testimonials = [
            { name: "Sara Ahmed", text: "Fastest booking ever. Driver was polite and arrived early!", avatar: "https://i.pravatar.cc/80?img=68" },
            { name: "Usman Ali", text: "Used for airport drop. Smooth ride, great price, highly recommend.", avatar: "https://i.pravatar.cc/80?img=44" },
            { name: "Ayesha Khan", text: "Family SUV was perfect for our trip. Very clean & spacious.", avatar: "https://i.pravatar.cc/80?img=32" }
        ];

        let current = 0;
        const carousel = document.getElementById('testimonial-carousel');

        function renderTestimonial() {
            carousel.innerHTML = `
                <div class="glass rounded-3xl p-10 text-center mx-auto max-w-3xl transition-all duration-700">
                    <img src="${testimonials[current].avatar}" alt="" class="w-20 h-20 rounded-full mx-auto mb-6 border-4 border-amber-500/30">
                    <p class="text-xl italic mb-6">"${testimonials[current].text}"</p>
                    <p class="font-bold text-amber-400">${testimonials[current].name}</p>
                </div>
            `;
        }

        renderTestimonial();
        setInterval(() => {
            current = (current + 1) % testimonials.length;
            renderTestimonial();
        }, 6000);

        // Accordion
        document.querySelectorAll('.accordion-header').forEach(header => {
            header.addEventListener('click', () => {
                const content = header.nextElementSibling;
                const isOpen = content.style.maxHeight;
                
                document.querySelectorAll('.accordion-content').forEach(c => {
                    if (c !== content) c.style.maxHeight = null;
                });

                content.style.maxHeight = isOpen ? null : content.scrollHeight + "px";
                header.querySelector('i').classList.toggle('rotate-180');
            });
        });
    </script>
